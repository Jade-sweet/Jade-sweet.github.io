数据库基础

类和类之间的关系（对象之间的关系）

  1. is-a关系：继承关系（泛化）
  2. has-a关系：关联关系（聚合关系、合成关系）- 重数
  3. use-a关系：依赖关系

数据库 - 数据的集散地（仓库）- 数据持久化（把数据长久的保存起来）
不仅可以保存数据，更重要的是能够很好的管理数据，方便将来对数据的检索，
大多数的数据库都能够保证数据的一致性、完整性并减少数据的冗余

数据库可以分为两大类：关系型数据库和非关系型数据库

1970s - IBM的研究员发表了论述关系模型的论文

关系型数据库的特点：

  1. 理论基础：集合论 + 关系代数
  2. 具体表象：用二维表来存储数据（行 - 记录、列 - 字段）
  3. 编程语言：SQL（结构化查询语言）
    • DDL（数据定义语言）：create / drop / alter
    • DML（数据操作语言）：insert / delete / update / select
    • DCL（数据控制语言）：grant / revoke

关系型数据库的产品：

  1. Oracle - 银行、电子政务、大型电商
  2. MySQL - 中小企业 —> MariaDB
  3. IBM DB2 / Microsoft SQL Server

使用MySQL管理数据
– 显示所有数据库
show databases;
– 如果存在名为school的数据库就删除它
drop database if exists school;
– 创建名为school的数据库并指定默认的字符集为utf-8
create database school default charset utf8;

-- 切换到school数据库
use school;
-- 查看所有表
show tables;
-- 如果存在名为tb_student的表就删除它
drop table if exists tb_student;

-- 创建名为tb_student的表
create table tb_student (
    stuid integer not null comment '学号',
    stuname varchar(20) not null comment '姓名',
    stusex boolean default 1 comment '性别',
    stubirth date comment '生日',
    primary key (stuid)
);

-- not null 非空约束（列的值不能为空）
-- defualt 默认值约束（如果没有给列赋值就使用默认值）
-- primary key 主键约束（唯一标识一条记录的列就是主键，必须唯一）


-- 修改tb_student表添加一个列
alter table tb_student add column stuaddr varchar(255);
-- 修改tb_student表删除一个列
alter table tb_student drop column stuaddr;
-- 修改tb_student表修改一个列
alter table tb_student modify column stuaddr varchar(100);
alter table tb_student change column stuaddr foo varchar(50);


-- 向表插入数据
insert into tb_student values (1001, '骆昊', 1, '1980-11-28', '四川成都');
insert into tb_student (stuid, stuname) values (1003, '王大锤');
insert into tb_student (stuid, stuname, stusex) values 
    (1004, '白元芳', 1), 
    (1005, '武则天', 0), 
    (1006, '狄仁杰', 1);
-- 删除数据
delete from tb_student where stuid=1002;
delete from tb_student where stuid in (1004, 1006);
truncate table tb_student;
-- 更新数据
update tb_student set stuaddr='四川绵阳' where stuid=1003;
update tb_student set stuaddr='四川广元', stubirth='1990-1-1' 
where stuname='武则天';


-- 查看tb_student表
select * from tb_student;
select * from tb_student\G

MySQL里面常用的数据类型
? data types; –这个语句是查看MySQL支持的数据类型
? varchar -> 长度可以指定为065535 用法：varchar(065535的数字)，表示此处为不定长字符，最长为()中的值

? integer -> 存放数字类型的数据
类型 储存空间 有符号最小值 无符号最小值 有符号最大值 无符号最大值
TINYINT 1 -128 0 127 255
SMALLINT 2 -32768 0 32767 65535
MEDIUMINT 3 -8388608 0 8388607 16777215
INT 4 -2147483648 0 2147483647 4294967295
BIGINT 8 -2^63 0 2^63-1 2^64-1 
? decimal  ->

在DECIMAL列声明中，可以（通常是）指定精度和小数位数。例如：salary DECIMAL(5,2)

在此示例中，5是精度， 2是小数位数。精度表示值存储的有效位数，小数位数表示小数点后可存储的位数。

? date  ->

该DATE类型用于具有日期部分但没有时间部分的值。MySQL检索并DATE以'YYYY-MM-DD'格式显示 值 。支持的范围是 '1000-01-01'到 '9999-12-31'

? datetime  ->

该DATETIME类型用于包含日期和时间部分的值。MySQL检索并DATETIME以‘YYYY-MM-DD hh:mm:ss’格式显示 值。支持的范围是 '1000-01-01 00:00:00'到'9999-12-31 23:59:59'

? char  -> 长度可以指定为0~255 用法：char(0~255的数字),表示此处为定长字符


-- 创建学院表并插入3-5条数据
drop table if exists tb_college;

create table tb_college (
    collid integer not null auto_increment comment '编号',
    collname varchar(50) not null comment '名称',
    collmaster varchar(20) default '' comment '院长',
    primary key (collid)
);

insert into tb_college (collname, collmaster) values
    ('计算机学院', '张三丰'),
    ('外国语学院', '任盈盈'),
    ('经济管理学院', '左冷禅');

-- 修改学生表，建立学生对学院的多对一关系
alter table tb_student add column collid integer;

-- 通过添加外键（foreign key）约束来建立学生表对学院表的参照关系（学生表中的学院编号要参照学院表的学院编号）
alter table tb_student add constraint fk_student_collid 
foreign key (collid) references tb_college (collid);

-- 错误：添加或更新操作违反了外键约束
ERROR 1452 (23000): Cannot add or update a child row: a foreign key constraint fails


-- 创建老师表tb_teacher
create table tb_teacher (
    teaid integer not null comment '工号',
    teaname varchar(20) not null comment '姓名',
    teatitle varchar(10) not null comment '职称',
    collid integer not null comment '所属学院',
    primary key (teaid),
    foreign key (collid) references tb_college (collid)
);


-- 创建课程表tb_course
create table tb_course (
    couid integer not null auto_increment comment '编号',
    couname varchar(50) not null comment '名称',
    coucredit integer not null comment '学分',
    teaid integer not null comment '授课老师',
    primary key (couid),
    foreign key (teaid) references tb_teacher (teaid)
);


-- 创建学生选课记录表
create table tb_record (
    recid integer not null auto_increment,
    sid integer not null,
    cid integer not null,
    seldate datetime default now(),
    score decimal(4, 1),
    primary key (recid)
);

-- 修改选课记录表添加外键约束
alter table tb_record add constraint fk_record_sid 
foreign key (sid) references tb_student (stuid);
alter table tb_record add constraint fk_record_cid 
foreign key (cid) references tb_course (couid);

-- 修改选课记录表添加唯一约束
alter table tb_record add constraint uni_record_sid_cid 
unique (sid, cid);
